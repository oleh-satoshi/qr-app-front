// Функция для извлечения параметров из URL function getQueryParams() { const params = new URLSearchParams(window.location.search); return { telegramId: params.get('telegramId'), totalScans: params.get('totalScans') || 0 }; } const ds = Telegram.WebApp.DeviceStorage; const params = getQueryParams(); if (params.telegramId) { // Сохраняем в DeviceStorage ds.setItem('telegramId', params.telegramId, () => {}); ds.setItem('totalScans', params.totalScans, () => {}); console.log('Saved to DeviceStorage:', params); document.getElementById('scanCount').textContent = params.totalScans; } else { // Загружаем из DeviceStorage ds.getItem('telegramId', (err, id) => { if (!err && id) { params.telegramId = id; } ds.getItem('totalScans', (err, scans) => { if (!err && scans) { params.totalScans = scans; document.getElementById('scanCount').textContent = scans; } else { document.getElementById('scanCount').textContent = 0; } console.log('Loaded from DeviceStorage:', params); }); }); }
